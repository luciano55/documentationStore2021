AddClient: Un proceso que lleva a cabo la inclusión de un cliente:

·         La tabla destino cclient

AddClient (myName, mySurname, myNifNie, myMobile, myEmail, myDateBird, myCP, myAddress, myAvatar, myUser, myPass, out message)

 

SET message = validateClient(myName, mySurname, myNifNie, myMobile, myEmail, myDateBird, myCP, myAvatar, myUser, myPass);

IF locate("Error",message) THEN SET message = concat(message, " -Error NO PASAMOS LA VALIDACIÓN");

   ELSE SET message = concat(message, " -OK PASAMOS LA VALIDACIÓN.");

    SET message = concat(message, verifyClient(myNifNie, myEmail, myCP, myUser));

    IF locate("Error",message) THEN SET message = concat(message, " -Error NO PASAMOS LA VERIFICACIÓN");

      ELSE SET message = concat(message, " -OK PASAMOS LA VERIFICACIÓN.");

           SET message = concat(message,  addClient(myName, mySurname, myNifNie, myMobile, myEmail, myDateBird, myCP, myAddress, myAvatar, myUser, myPass));

     END IF; 

END IF;
